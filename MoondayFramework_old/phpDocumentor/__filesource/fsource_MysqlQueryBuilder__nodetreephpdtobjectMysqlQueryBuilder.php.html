  58.     function select_query( $aParams, $aFields ) {
    59. 

  59.         $aParams     = $this->_keystoupper( $aParams );
    60. 

  60.         $sAction     = 'SELECT';
    61. 

  61.         $sFields     = isset( $aParams['SELECT'] ) ? $aParams['SELECT'] : implode( ", ", $aFields );
    62. 

  62.         $sFrom         = isset( $aParams['FROM'] ) ? 'FROM `'.$aParams['FROM'].'`' : 'FROM `'.$this->sTargetTable.'`';
    63. 

  63.         $sWhere     = isset( $aParams['WHERE'] ) ? 'WHERE '.$aParams['WHERE'] : 'WHERE 1';
    64. 

  64.         $sGroupBy     = isset( $aParams['GROUP BY'] ) ? 'GROUP BY '.$aParams['GROUP BY'] : '';
    65. 

  65.         $sHaving     = isset( $aParams['HAVING'] ) ? 'HAVING '.$aParams['HAVING'] : '';
    66. 

  66.         $sOrderBy     = isset( $aParams['ORDER BY'] ) ? 'ORDER BY '.$aParams['ORDER BY'] : '';
    67. 

  67.         $sLimit     = isset( $aParams['LIMIT'] ) ? 'LIMIT '.$aParams['LIMIT'] : '';
    68. 

  68.         $sOffset     = isset( $aParams['OFFSET'] ) ? 'OFFSET '.$aParams['OFFSET'] : '';
    69. 

  69.         $sProcedure = isset( $aParams['PROCEDURE'] ) ? 'PROCEDURE '.$aParams['PROCEDURE'] : '';
    70. 

  70.         
    71. 

  71.         $sQuery     = "$sAction $sFields $sFrom $sWhere $sGroupBy $sHaving $sOrderBy $sLimit $sOffset $sProcedure";

  152.      * Creates set clause for update and insert

  157.     function _get_set_clause( $aFields ) {
    158. 

  158.         $sSetClause = '';
    159. 

  159.         foreach( $aFields as $key => $val ) {
    160. 

  160.             switch( gettype( $val ) ) {
    161. 

  161.                 case 'boolean':
    162. 

  162.                     $val = $val ? 1 : 0;
    163. 

  163.                     break;
    164. 

  164.                 case 'NULL':
    165. 

  165.                     $val = 'NULL';
    166. 

  166.                     break;
    167. 

  167.                 case 'string':
    168. 

  168.                     $val = "'$val'";
    169. 

  169.                     break;
    170. 

  170.                 case 'integer':
    171. 

  171.                     $val = $val;
    172. 

  172.                     break;
    173. 

  173.                 case 'double':
    174. 

  174.                     $val = $val;
    175. 

  175.                     break;
    176. 

  176.                 case 'array':
    177. 

  177.                     $val = "'".implode( ',', $val )."'";
    178. 

  178.                     break;
    179. 

  179.                 case 'object':
    180. 

  180.                     $val = "'".addslashes( serialize( $val ) )."'";
    181. 

  181.                     break;
    182. 

  182.                 default:
    183. 

  183.                     $val = "'$val'";
    184. 

  184.                     break;
    185. 

  185.             }
    186. 

  186.             $sSetClause .= "`".$key."` = $val, ";
    187. 

  187.         }
    188. 

  188.         return substr( $sSetClause, 0, strlen( $sSetClause ) - 2 );
    189. 

  189.     }
