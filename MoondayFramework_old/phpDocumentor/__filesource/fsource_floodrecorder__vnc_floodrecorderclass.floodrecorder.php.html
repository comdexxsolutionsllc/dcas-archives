  54.     function isBlock($index = null,$access=-1,$interval=-1,$time=-1) {
    55. 

  55.         if ($access == -1) $access = $this->maxaccess;
    56. 

  56.         if ($interval == -1) $interval = $this->interval;
    57. 

  57.         if ($time == -1) $time = $this->blocktime;
    58. 

  58.         
    59. 

  59.         $path = $this->_path($index);
    60. 

  60.         if (file_exists($path . "/block")) {
    61. 

  61.             //this ip has been blocked , let take a small check
    62. 

  62.       $time = $this->_readfile("$path/block");
    63. 

  63.       if (time() > $time) $this->_unlink("$path/block");
    64. 

  64.       else return true;
    65. 

  65.         }
    66. 

  66.         $block = false;
    67. 

  67.         if ($dh = opendir($path)) {
    68. 

  68.                 $count = 0;
    69. 

  69.         while (($file = readdir($dh)) !== false) {
    70. 

  70.             if ($file == '.' || $file == '..') continue;
    71. 

  71.             if (is_numeric($file)) {
    72. 

  72.                 if ($file < (time() - $interval)) {
    73. 

  73.                     //old file , just delete it
    74. 

  74.                     $this->_unlink("$path/$file");
    75. 

  75.                     continue;
    76. 

  76.                 }
    77. 

  77.                 //access counter
    78. 

  78.                 $count++;
    79. 

  79.             }
    80. 

  80.         }
    81. 

  81.         closedir($dh);
    82. 

  82.         if ($count >= $access) {
    83. 

  83.             $this->block($index , $time);
    84. 

  84.             $block = true;
    85. 

  85.         }
    86. 

  86.     }

  87.     return $block;
