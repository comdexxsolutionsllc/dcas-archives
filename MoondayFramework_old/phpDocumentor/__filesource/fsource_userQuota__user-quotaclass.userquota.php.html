  161.     function _htmlHexToBinArray($hex)
    162. 

  162.     {
    163. 

  163.         for ($i=0; $i<3; $i++)
    164. 

  164.         {
    165. 

  165.             $foo = substr($hex, 2*$i, 2); 
    166. 

  166.             $rgb[$i] = 16 * hexdec(substr($foo, 0, 1)) + hexdec(substr($foo, 1, 1)); 
    167. 

  167.         }
    168. 

  168.         return $rgb;

  189.     function _displayImage()
    190. 

  190.     {
    191. 

  191.         // the graph variables
    192. 

  192.         $sStart = $this->g3DHeight * 2;
    193. 

  193.         $wStart = $this->gWidth * 2;
    194. 

  194.         $hStart = $this->gHeight * 2;
    195. 

  195.         if ($this->qUsed >= $this->qQuota) $usedPercent = 360;
    196. 

  196.         else $usedPercent = $this->oPercent * 3.6;
    197. 

  197.  
    198. 

  198.         // work out where the 'used' wedge will be located
    199. 

  199.         if ($this->gCentreUsed)
    200. 

  200.         {
    201. 

  201.             $sWedge = (int)(90 - ($usedPercent / 2));
    202. 

  202.             if ($sWedge < 0) $sWedge += 360;
    203. 

  203.             $mWedge = (int)(90 + ($usedPercent / 2));
    204. 

  204.             $mWedge = ($mWedge == 90 ? 91 : $mWedge);
    205. 

  205.             $eWedge = (int)$sWedge;
    206. 

  206.         }
    207. 

  207.         else
    208. 

  208.         {
    209. 

  209.             $sWedge = 0;
    210. 

  210.             $mWedge = (int)($usedPercent ? $usedPercent : 1);
    211. 

  211.             $eWedge = 360;
    212. 

  212.         }
    213. 

  213.  
    214. 

  214.         // setup image and main colours
    215. 

  215.         $im = @ImageCreateTrueColor($wStart, $hStart + $sStart);
    216. 

  216.         if ($im)
    217. 

  217.         {
    218. 

  218.             $cBg = ImageColorAllocate($im, $this->gBackColour[0], $this->gBackColour[1], $this->gBackColour[2]);
    219. 

  219.             $cUsed = ImageColorAllocate($im, $this->gUsedColour[0], $this->gUsedColour[1], $this->gUsedColour[2]);
    220. 

  220.             $cQuota = ImageColorAllocate($im, $this->gQuotaColour[0], $this->gQuotaColour[1], $this->gQuotaColour[2]);
    221. 

  221.             ImageFill($im, 0, 0, $cBg);
    222. 

  222.  
    223. 

  223.             // work out 3D look if needs be
    224. 

  224.             if ($this->g3DHeight)
    225. 

  225.             {
    226. 

  226.                 // process colours
    227. 

  227.                 $qDarkArray = $this->gQuotaColour;
    228. 

  228.                 for ($i=0; $i<3; $i++)
    229. 

  229.                 {
    230. 

  230.                     ($qDarkArray[$i] > 99) ? $qDarkArray[$i] -= 100 : $qDarkArray[$i] = 0;
    231. 

  231.                 }
    232. 

  232.                 $uDarkArray = $this->gUsedColour;
    233. 

  233.                 for ($i=0; $i<3; $i++)
    234. 

  234.                 {
    235. 

  235.                     ($uDarkArray[$i] > 99) ? $uDarkArray[$i] -= 100 : $uDarkArray[$i] = 0;
    236. 

  236.                 }
    237. 

  237.                 $cQuotaDark = ImageColorAllocate($im, $qDarkArray[0], $qDarkArray[1], $qDarkArray[2]);
    238. 

  238.                 $cUsedDark = ImageColorAllocate($im, $uDarkArray[0], $uDarkArray[1], $uDarkArray[2]);
    239. 

  239.  
    240. 

  240.                 // add 3D look
    241. 

  241.                 $shadow_start = ($hStart/2) + $sStart;
    242. 

  242.                 $shadow_end = $hStart/2;
    243. 

  243.                 for ($i=$shadow_start; $i>$shadow_end; $i--)
    244. 

  244.                 {
    245. 

  245.                     ImageFilledArc($im, ($wStart/2), $i, $wStart, $hStart,  $sWedge, $mWedge, $cUsedDark, IMG_ARC_PIE);
    246. 

  246.                     ImageFilledArc($im, ($wStart/2), $i, $wStart, $hStart, $mWedge, $eWedge, $cQuotaDark, IMG_ARC_PIE);
    247. 

  247.                 }
    248. 

  248.             }
    249. 

  249.  
    250. 

  250.             // now do the top of the graph
    251. 

  251.             ImageFilledArc($im, ($wStart/2), ($hStart/2), $wStart, $hStart, $sWedge, $mWedge, $cUsed, IMG_ARC_PIE);
    252. 

  252.             ImageFilledArc($im, ($wStart/2), ($hStart/2), $wStart, $hStart, $mWedge, $eWedge, $cQuota, IMG_ARC_PIE);
    253. 

  253.  
    254. 

  254.             // now create a legend image if needs be
    255. 

  255.             if ($this->gLegend)
    256. 

  256.             {
    257. 

  257.                 // the legend variables
    258. 

  258.                 $lHeight = $lWidth = 0;
    259. 

  259.                 $spacer = 10;
    260. 

  260.  
    261. 

  261.                 // build quota strings
    262. 

  262.                 $qText[0] = 'Quota: ' . $this->_formatSize($this->qQuota);
    263. 

  263.                 if ($this->qUser) $qText[1] = '       (' . $this->qUser . ')';
    264. 

  264.                 else $qText[1] = '';
    265. 

  265.                 $uText[0] = 'Used : ' . $this->_formatSize($this->qUsed);
    266. 

  266.                 $uText[1] = '       (' . number_format($this->oPercent,2) . '%)';
    267. 

  267.  
    268. 

  268.                 // space + line + spacer + line
    269. 

  269.                 $lHeight = (ImageFontHeight(2) * ($qText[1] == '' ? 3 : 4)) + $spacer;
    270. 

  270.  
    271. 

  271.                 // get biggest string length and add spacer to it - legend block is size of font height (square)
    272. 

  272.                 $qMax = (strlen($qText[0]) > strlen($qText[1])) ? strlen($qText[0]) : strlen($qText[1]);
    273. 

  273.                 $uMax = (strlen($uText[0]) > strlen($uText[1])) ? strlen($uText[0]) : strlen($uText[1]);
    274. 

  274.                 $tMax = ($qMax > $uMax ? $qMax : $uMax);
    275. 

  275.                 $lWidth = ($tMax * ImageFontWidth(2)) + $spacer + ImageFontHeight(2);
    276. 

  276.  
    277. 

  277.                 // now create the image
    278. 

  278.                 $lim = ImageCreateTrueColor($lWidth, $lHeight);
    279. 

  279.                 ImageFill($lim, 0, 0, $cBg);
    280. 

  280.                 $cText = ImageColorAllocate($lim, $this->gTextColour[0], $this->gTextColour[1], $this->gTextColour[2]);
    281. 

  281.                 $lx = 0;
    282. 

  282.                 $ly = 0;
    283. 

  283.  
    284. 

  284.                 // write out the 'quota' legend
    285. 

  285.                 ImageFilledRectangle($lim, $lx, $ly, ($lx + ImageFontHeight(2)), ($ly + ImageFontHeight(2)), $cQuota);
    286. 

  286.                 ImageString($lim, 2, ($lx + ImageFontHeight(2) + $spacer), $ly, $qText[0], $cText);
    287. 

  287.                 if ($qText[1] != '')
    288. 

  288.                 {
    289. 

  289.                     $ly += ImageFontHeight(2);
    290. 

  290.                     ImageString($lim, 2, ($lx + ImageFontHeight(2) + $spacer), $ly, $qText[1], $cText);
    291. 

  291.                 }
    292. 

  292.  
    293. 

  293.                 $ly += ($spacer + ImageFontHeight(2));
    294. 

  294.  
    295. 

  295.                 // write out the 'used' legend
    296. 

  296.                 ImageFilledRectangle($lim, $lx, $ly, ($lx + ImageFontHeight(2)), ($ly + ImageFontHeight(2)), $cUsed);
    297. 

  297.                 ImageString($lim, 2, ($lx + ImageFontHeight(2) + $spacer), $ly, $uText[0], $cText);
    298. 

  298.                 $ly += ImageFontHeight(2);
    299. 

  299.                 ImageString($lim, 2, ($lx + ImageFontHeight(2) + $spacer), $ly, $uText[1], $cText);
    300. 

  300.  
    301. 

  301.                 // now merge the two images into the final one
    302. 

  302.  
    303. 

  303.                 // anti-aliasing look
    304. 

  304.                 $gsx = ImageSX($im);
    305. 

  305.                 $gsy = ImageSY($im);
    306. 

  306.                 $lsx = ImageSX($lim);
    307. 

  307.                 $lsy = ImageSY($lim);
    308. 

  308.                 $gnx = ($gsx >> 1);
    309. 

  309.                 $gny = ($gsy >> 1);
    310. 

  310.                 $fx = ($gnx > $lsx) ? $gnx : $lsx;
    311. 

  311.                 $fy = $gny + $lsy + ($spacer * 2);
    312. 

  312.                 $final = ImageCreateTrueColor($fx, $fy);
    313. 

  313.                 ImageFill($final, 0, 0, $cBg);
    314. 

  314.                 ImageCopyResampled($final, $im, (($fx/2)-($gnx/2)), 0, 0, 0, $gnx, $gny, $gsx, $gsy);
    315. 

  315.                 ImageCopyResampled($final, $lim, (($fx/2)-($lsx/2)), $gny + ($spacer * 2), 0, 0, $lsx, $lsy, $lsx, $lsy);
    316. 

  316.                 ImageDestroy($lim);
    317. 

  317.             }
    318. 

  318.             else
    319. 

  319.             {
    320. 

  320.                 // we do not have a legend, so just reample graph
    321. 

  321.                 $sx = ImageSX($im); 
    322. 

  322.                 $sy = ImageSY($im); 
    323. 

  323.                 $nx = ($sx>>1); 
    324. 

  324.                 $ny = ($sy>>1);
    325. 

  325.                 $final = ImageCreateTrueColor($nx, $ny);
    326. 

  326.                 ImageCopyResampled($final, $im, 0, 0, 0, 0, $nx, $ny, $sx, $sy);
    327. 

  327.             }
    328. 

  328.  
    329. 

  329.             // flush image
    330. 

  330.             header("Content-type: image/jpeg");
    331. 

  331.             ImageJPEG($final, NULL, 100);
    332. 

  332.             ImageDestroy($im);
    333. 

  333.             ImageDestroy($final);
    334. 

  334.         }
    335. 

  335.     }

  341.     function _formatSize($size = 0)
    342. 

  342.     {
    343. 

  343.         if ($size >= 1073741824) $size = round($size/1073741824*100)/100 . " Gb";
    344. 

  344.         else if ($size >= 1048576) $size = round($size/1048576*100)/100 . " Mb";
    345. 

  345.         else if ($size >= 1024) $size = round($size/1024*100)/100 . " kb";
    346. 

  346.         else $size = $size . " bytes";
    347. 

  347.         return $size;
    348. 

  348.     }
