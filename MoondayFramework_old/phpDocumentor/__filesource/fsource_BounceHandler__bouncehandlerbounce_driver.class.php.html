  84.     function standard_parser($content){ // associative array orstr
    85. 

  85.         // receives email head as array of lines
    86. 

  86.         // simple parse (Entity: value\n)

  87.         if(!is_array($content)) $content = explode("\r\n", $content);
    88. 

  88.         foreach($content as $line){
    89. 

  89.             if(preg_match('/([^\s.]*):\s(.*)/', $line, $array)){
    90. 

  90.                 $entity = ucfirst(strtolower($array[1]));
    91. 

  91.                 if(! $hash[$entity]){
    92. 

  92.                     $hash[$entity] = $array[2];
    93. 

  93.                 }
    94. 

  94.                 else if($hash['Received']){

    99.

  100.                     }

  101.                 }
