  133.         function sendQuery( $dnsquery, $useTCP = false ) {
    134. 

  134.             $answer = false;
    135. 

  135.             $out_buf = $dnsquery->asOctets( false );
    136. 

  136.             $out_buf_len = strlen( $out_buf );
    137. 

  137.             if( $out_buf ) {
    138. 

  138.                 if( $useTCP == false && $out_buf_len <= DNS_UDP_PACKET_MAX_LENGTH ) {
    139. 

  139.                     /* connection by UDP */
    140. 

  140.                     if( ( $sock = fsockopen( 'udp://'.$this->nameserver, $this->port, $this->timeout ) ) === false ) {
    141. 

  141.                         return false;
    142. 

  142.                     }
    143. 

  143.                     socket_set_blocking( $sock, true );
    144. 

  144.                     if( fwrite( $sock, $out_buf ) == $out_buf_len ) {
    145. 

  145.                         $answer = new DNSAnswer( $sock, DNS_UDP_PACKET_MAX_LENGTH );
    146. 

  146.                     }
    147. 

  147.                     fclose( $sock );
    148. 

  148.                 } else {
    149. 

  149.                     /* connection by TCP */
    150. 

  150.                     if( ( $sock = fsockopen( $this->nameserver, $this->port, $this->timeout ) ) === false ) {
    151. 

  151.                         return false;
    152. 

  152.                     }
    153. 

  153.                     socket_set_blocking( $sock, true );
    154. 

  154.                     if( fwrite( $sock, pack( 'n', $out_buf_len ) ) == 2 &&
    155. 

  155.                         fwrite( $sock, $out_buf ) == $out_buf_len ) {
    156. 

  156.                         $tmp = unpack( 'nl', fread( $sock, 2 ) );
    157. 

  157.                         $limit_length = $tmp['l'];
    158. 

  158.                         print $limit_length;
    159. 

  159.                         $answer = new DNSAnswer( $sock, $limit_length );
    160. 

  160.                     }
    161. 

  161.                     fclose( $sock );
    162. 

  162.                 }
    163. 

  163.             }
    164. 

  164.             return $answer;
    165. 

  165.         }

  176.         function DNSQuery( &$dnsrecord, $flags = DNS_HEADERSPEC_RECURSION_DESIRED )
    177. 

  177.         {
    178. 

  178.             $this->id = rand( 1, 255 ) | ( rand( 0, 255 ) << 8 );

  179.             $this->flags = $flags & DNS_HEADERSPEC_QUERY_SPEC_MASK;

  183.         function asOctets() {
    184. 

  184.             if( $this->query_record->name === false ) { return false; }
    185. 

  185.             $buf = '';
    186. 

  186.             $buf .= pack( "nnnnnn", $this->id, DNS_OPCODE_QUERY | $this->flags, 1, 0, 0, 0 );
    187. 

  187.             $buf .= $this->query_record->name->asOctets();
    188. 

  188.             $buf .= pack( "nn", $this->query_record->type, $this->query_record->dclass );
    189. 

  189.             return $buf;
    190. 

  190.         }

  206.         function readStreamDirectly( $nbytes ) {
    207. 

  207.             if( ( $this->limit -= $nbytes ) < 0 ) {
    208. 

  208.                 return false;
    209. 

  209.             }
    210. 

  210.             $buf = fread( $this->stream, $nbytes );
    211. 

  211.             $this->octets .= $buf;

  216.         function readBufferedStream( $nbytes, $offset ) {
    217. 

  217.             if( $offset < $this->nbytes_read ) {
    218. 

  218.                 $diff = $this->nbytes_read - $offset;
    219. 

  219.                 if( $nbytes <= $diff ) {
    220. 

  220.                     return substr( $this->octets, $offset, $nbytes );
    221. 

  221.                 } else {
    222. 

  222.                     $buf = substr( $this->octets, $offset, $diff );
    223. 

  223.                     $nbytes -= $diff;
    224. 

  224.                 }
    225. 

  225.             } else {
    226. 

  226.                 $buf = '';
    227. 

  227.                 while( $offset > $this->nbytes_read ) {
    228. 

  228.                     if( $this->readStreamDirectly( $offset - $this->nbytes_read ) === false ) {
    229. 

  229.                         return false;
    230. 

  230.                     }
    231. 

  231.                 }
    232. 

  232.             }
    233. 

  233.             if( ( $_buf = $this->readStreamDirectly( $nbytes ) ) === false ) {
    234. 

  234.                 return false;
    235. 

  235.             }
    236. 

  236.             $buf .= $_buf;
    237. 

  237.             return $buf;
    238. 

  238.         }

  357.         function getLabels( $max_length = 255, $offset = -1 ) {
    358. 

  358.             if( $offset < 0 ) { $offset = $this->nbytes_read; }
    359. 

  359.             $labels = array();
    360. 

  360.             for(;;) {
    361. 

  361.                 if( --$max_length < 0 ) { return false; }
    362. 

  362.                 if( ( $buf = $this->readBufferedStream( 1, $offset ) ) === false ) {
    363. 

  363.                     return false;
    364. 

  364.                 }
    365. 

  365.                 $label_len = ord( $buf );
    366. 

  366.                 ++$offset;
    367. 

  367.                 if( $label_len < 64 ) {
    368. 

  368.                     /* uncompressed */
    369. 

  369.                     if( ( $max_length -= $label_len ) < 0 ) { return false; }
    370. 

  370.                     if( ( $labels[] = $this->readBufferedStream( $label_len, $offset ) ) === false ) {
    371. 

  371.                         return false;
    372. 

  372.                     }
    373. 

  373.                     $offset += $label_len;
    374. 

  374.                     if( $label_len == 0 ) { break; }
    375. 

  375.                 } else {
    376. 

  376.                     /* compressed */
    377. 

  377.                     if( ( $buf = $this->readBufferedStream( 1, $offset ) ) === false ) {
    378. 

  378.                         return false;
    379. 

  379.                     }
    380. 

  380.                     if( --$max_length < 0 ) {
    381. 

  381.                         return false;
    382. 

  382.                     }
    383. 

  383.                     $_offset = ( ( $label_len & 0x3f ) << 8 ) + ord( $buf );
    384. 

  384.  
    385. 

  385.                     if( ($_labels = $this->getLabels( $offset - $_offset, $_offset )) === false ) {
    386. 

  386.                         return false;
    387. 

  387.                     }
    388. 

  388.                     $labels = array_merge( $labels, $_labels );
    389. 

  389.                     break;
    390. 

  390.                 }
    391. 

  391.             }
    392. 

  392.             return $labels;
    393. 

  393.         }
