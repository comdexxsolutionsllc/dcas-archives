  36.     Function addFile($filename, $cfilename, $fileComments='', $data=false){
    37. 

  37.         if(!($fh = &$this->fh))
    38. 

  38.             $fh = fopen($this->filename, $this->overwrite?'wb':'a+b');
    39. 

  39.         
    40. 

  40.         // $filename can be a local file OR the data wich will be compressed
    41. 

  41.         if(substr($cfilename, -1)=='/'){
    42. 

  42.             $details['uncsize'] = 0;
    43. 

  43.             $data = '';
    44. 

  44.         }
    45. 

  45.         elseif(file_exists($filename)){
    46. 

  46.             $details['uncsize'] = filesize($filename);
    47. 

  47.             $data = file_get_contents($filename);
    48. 

  48.         }
    49. 

  49.         elseif($filename){
    50. 

  50.             echo "<b>Cannot add $filename. File not found</b><br>";
    51. 

  51.             return false;
    52. 

  52.         }
    53. 

  53.         else{
    54. 

  54.             $details['uncsize'] = strlen($filename);
    55. 

  55.             // DATA is given.. use it! :|
    56. 

  56.         }
    57. 

  57.         
    58. 

  58.         // if data to compress is too small, just store it
    59. 

  59.         if($details['uncsize'] < 256){
    60. 

  60.             $details['comsize'] = $details['uncsize'];
    61. 

  61.             $details['vneeded'] = 10;
    62. 

  62.             $details['cmethod'] = 0;
    63. 

  63.             $zdata = &$data;
    64. 

  64.         }
    65. 

  65.         else{ // otherwise, compress it
    66. 

  66.             $zdata = gzcompress($data);
    67. 

  67.             $zdata = substr(substr($zdata, 0, strlen($zdata) - 4), 2); // fix crc bug (thanks to Eric Mueller)
    68. 

  68.             $details['comsize'] = strlen($zdata);
    69. 

  69.             $details['vneeded'] = 10;
    70. 

  70.             $details['cmethod'] = 8;
    71. 

  71.         }
    72. 

  72.         
    73. 

  73.         $details['bitflag'] = 0;
    74. 

  74.         $details['crc_32']  = crc32($data);
    75. 

  75.         
    76. 

  76.         // Convert date and time to DOS Format, and set then
    77. 

  77.         $lastmod_timeS  = str_pad(decbin(date('s')>=32?date('s')-32:date('s')), 5, '0', STR_PAD_LEFT);
    78. 

  78.         $lastmod_timeM  = str_pad(decbin(date('i')), 6, '0', STR_PAD_LEFT);
    79. 

  79.         $lastmod_timeH  = str_pad(decbin(date('H')), 5, '0', STR_PAD_LEFT);
    80. 

  80.         $lastmod_dateD  = str_pad(decbin(date('d')), 5, '0', STR_PAD_LEFT);
    81. 

  81.         $lastmod_dateM  = str_pad(decbin(date('m')), 4, '0', STR_PAD_LEFT);
    82. 

  82.         $lastmod_dateY  = str_pad(decbin(date('Y')-1980), 7, '0', STR_PAD_LEFT);
    83. 

  83.         
    84. 

  84.         # echo "ModTime: $lastmod_timeS-$lastmod_timeM-$lastmod_timeH (".date("s H H").")\n";
    85. 

  85.         # echo "ModDate: $lastmod_dateD-$lastmod_dateM-$lastmod_dateY (".date("d m Y").")\n";
    86. 

  86.         $details['modtime'] = bindec("$lastmod_timeH$lastmod_timeM$lastmod_timeS");
    87. 

  87.         $details['moddate'] = bindec("$lastmod_dateY$lastmod_dateM$lastmod_dateD");
    88. 

  88.         
    89. 

  89.         $details['offset'] = ftell($fh);
    90. 

  90.         fwrite($fh, $this->zipSignature);
    91. 

  91.         fwrite($fh, pack('s', $details['vneeded'])); // version_needed
    92. 

  92.         fwrite($fh, pack('s', $details['bitflag'])); // general_bit_flag
    93. 

  93.         fwrite($fh, pack('s', $details['cmethod'])); // compression_method
    94. 

  94.         fwrite($fh, pack('s', $details['modtime'])); // lastmod_time
    95. 

  95.         fwrite($fh, pack('s', $details['moddate'])); // lastmod_date
    96. 

  96.         fwrite($fh, pack('V', $details['crc_32']));  // crc-32
    97. 

  97.         fwrite($fh, pack('I', $details['comsize'])); // compressed_size
    98. 

  98.         fwrite($fh, pack('I', $details['uncsize'])); // uncompressed_size
    99. 

  99.         fwrite($fh, pack('s', strlen($cfilename)));   // file_name_length
    100. 

  100.         fwrite($fh, pack('s', 0));  // extra_field_length
    101. 

  101.         fwrite($fh, $cfilename);    // file_name
    102. 

  102.         // ignoring extra_field
    103. 

  103.         fwrite($fh, $zdata);
    104. 

  104.         
    105. 

  105.         // Append it to central dir
    106. 

  106.         $details['external_attributes']  = (substr($cfilename, -1)=='/'&&!$zdata)?16:32; // Directory or file name
    107. 

  107.         $details['comments']             = $fileComments;
    108. 

  108.         $this->appendCentralDir($cfilename, $details);
    109. 

  109.         $this->files_count++;
    110. 

  110.     }

  114.     Function save($zipComments=''){
    115. 

  115.         if(!($fh = &$this->fh))
    116. 

  116.             $fh = fopen($this->filename, $this->overwrite?'w':'a+');
    117. 

  117.         
    118. 

  118.         $cdrec = "";
    119. 

  119.         foreach($this->centraldirs as $filename=>$cd){
    120. 

  120.             $cdrec .= $this->dirSignature;
    121. 

  121.             $cdrec .= "\x0\x0";                  // version made by
    122. 

  122.             $cdrec .= pack('v', $cd['vneeded']); // version needed to extract
    123. 

  123.             $cdrec .= "\x0\x0";                  // general bit flag
    124. 

  124.             $cdrec .= pack('v', $cd['cmethod']); // compression method
    125. 

  125.             $cdrec .= pack('v', $cd['modtime']); // lastmod time
    126. 

  126.             $cdrec .= pack('v', $cd['moddate']); // lastmod date
    127. 

  127.             $cdrec .= pack('V', $cd['crc_32']);  // crc32
    128. 

  128.             $cdrec .= pack('V', $cd['comsize']); // compressed filesize
    129. 

  129.             $cdrec .= pack('V', $cd['uncsize']); // uncompressed filesize
    130. 

  130.             $cdrec .= pack('v', strlen($filename)); // file comment length
    131. 

  131.             $cdrec .= pack('v', 0);                // extra field length
    132. 

  132.             $cdrec .= pack('v', strlen($cd['comments'])); // file comment length
    133. 

  133.             $cdrec .= pack('v', 0); // disk number start
    134. 

  134.             $cdrec .= pack('v', 0); // internal file attributes
    135. 

  135.             $cdrec .= pack('V', $cd['external_attributes']); // internal file attributes
    136. 

  136.             $cdrec .= pack('V', $cd['offset']); // relative offset of local header
    137. 

  137.             $cdrec .= $filename;
    138. 

  138.             $cdrec .= $cd['comments'];
    139. 

  139.         }
    140. 

  140.         $before_cd = ftell($fh);
    141. 

  141.         fwrite($fh, $cdrec);
    142. 

  142.         
    143. 

  143.         // end of central dir
    144. 

  144.         fwrite($fh, $this->dirSignatureE);
    145. 

  145.         fwrite($fh, pack('v', 0)); // number of this disk
    146. 

  146.         fwrite($fh, pack('v', 0)); // number of the disk with the start of the central directory
    147. 

  147.         fwrite($fh, pack('v', $this->files_count)); // total # of entries "on this disk" 
    148. 

  148.         fwrite($fh, pack('v', $this->files_count)); // total # of entries overall 
    149. 

  149.         fwrite($fh, pack('V', strlen($cdrec)));     // size of central dir 
    150. 

  150.         fwrite($fh, pack('V', $before_cd));         // offset to start of central dir
    151. 

  151.         fwrite($fh, pack('v', strlen($zipComments))); // .zip file comment length
    152. 

  152.         fwrite($fh, $zipComments);
    153. 

  153.         
    154. 

  154.         fclose($fh);
    155. 

  155.     }
