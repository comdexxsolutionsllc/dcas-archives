  335.     function DrawGraph($file = "") {

    343.

    345. 

  345.       $this->__PaintBackground();

  351.         $this->__DrawBarText($p, $name);

  352.         $this->__DrawBar($p, $value);
    353. 

  353.       }
    354. 

  354.  
    355. 

  355.       if (strlen($this->graph_title)) {
    356. 

  356.         $this->__AllocateColor("im_graph_titlecolor",
    357. 

  357.                                $this->graph_titlecolor,
    358. 

  358.                                $this->graph_transparencylevel);
    359. 

  359.         $this->__DrawText($this->graph_title,
    360. 

  360.                           floor($this->graph_width / 2),
    361. 

  361.                           $this->graph_borderwidth + 2,
    362. 

  362.                           $this->im_graph_titlecolor,
    363. 

  363.                           2,
    364. 

  364.                           1);
    365. 

  365.       }
    366. 

  366.       
    367. 

  367.       if (strlen($file)) {
    368.

  372.         $ret = true;
    373. 

  373.       }
    374.

    375.

    376. 

  376.     }
