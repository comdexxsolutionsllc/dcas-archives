  154.           # parameters

  156.           $default_source = $this->text_root . 'lipsum.inc';
    157. 

  157.

  160.           
    161. 

  161.           # regex
    162. 

  162.           $_REGEX['stop_at_next_sent'] = '#([\.\?\!]+["\']*\s+\w)#U';

  163.           $_REGEX['start_at_next_sent'] = '#([\.\?\!]+["\']*\s+\w)#U';

    165. 

  165.           # flags

  166.           $_FLAG['loop_seam_found'] = 0;

  167.       
    168.

    169. 

  169.           $sample = '';

    171. 

  171.         
    172. 

  172.         // MANIPULATE          
    173. 

  173.                       
    174. 

  174.           # sanity check
    175. 

  175.           if ( !is_file($source_fpath) )
    176. 

  176.           {
    177. 

  177.               trigger_error("file [$source_fpath] not found", E_USER_NOTICE);
    178. 

  178.               return FALSE;
    179. 

  179.           }
    180. 

  180.           
    181. 

  181.           # check num words
    182. 

  182.           if ( $num_words > $max_sample_length ) 
    183. 

  183.           {
    184. 

  184.               trigger_error('maximum num of words in sample is 500, sample length will be 500 words', E_USER_NOTICE); 
    185. 

  185.               $num_words = $max_sample_length; 
    186. 

  186.           }
    187. 

  187.       
    188. 

  188.           # rip file
    189. 

  189.           $_raw_source = $this->_databug_rip_file($source_fpath);
    190. 

  190.           $_raw_source = strip_tags(trim($_raw_source));
    191. 

  191.           
    192. 

  192.           # strlen (PHP)
    193. 

  193.           $_strlen = strlen($_raw_source);
    194. 

  194.           
    195. 

  195.           # random strpos
    196. 

  196.           $_start_pos = mt_rand(1, $_strlen);
    197. 

  197.           
    198. 

  198.           # find start of next sentence
    199. 

  199.           
    200. 

  200.               # note: try 4 times (in case beginning at end of file)
    201. 

  201.               $_num_tries = 0;
    202. 

  202.               
    203. 

  203.               while ( $_num_tries <= 4 && !$_FLAG['loop_seam_found'] )
    204. 

  204.               {
    205. 

  205.                   $_num_tries++;
    206. 

  206.                   if ( preg_match($_REGEX['stop_at_next_sent'], $_raw_source, $MATCH, PREG_OFFSET_CAPTURE, $_start_pos) )
    207. 

  207.                   {
    208. 

  208.                       $_FLAG['loop_seam_found'] = 1;
    209. 

  209.                       $_loop_seam = $MATCH[1][1] + strlen($MATCH[1][0] - 1);
    210. 

  210.                   }
    211. 

  211.                   
    212. 

  212.                   # try new start position
    213. 

  213.                   $_start_pos = mt_rand(1, $_strlen);
    214. 

  214.                   
    215. 

  215.                   # reset to beginning of string on try 3
    216. 

  216.                   if ( $_num_tries == 3 )
    217. 

  217.                   {
    218. 

  218.                       $_start_pos = 0;
    219. 

  219.                   }
    220. 

  220.               }
    221. 

  221.               
    222. 

  222.               # DEBUG
    223. 

  223.               #trigger_notice("$_start_pos -> $_loop_seam", 'START POS -> LOOP SEAM');
    224. 

  224.               #trigger_notice($MATCH[1][0]);
    225. 

  225.                   
    226. 

  226.               # check
    227. 

  227.               if ( !$_FLAG['loop_seam_found'] )
    228. 

  228.               {
    229. 

  229.                   trigger_error('preg match failed', E_USER_WARNING);
    230. 

  230.                   return FALSE;
    231. 

  231.               }
    232. 

  232.           
    233. 

  233.           # snakeloop text
    234. 

  234.           $_snake_source_pt1 = substr($_raw_source, $_loop_seam) . "\n\n";
    235. 

  235.           $_snake_source_pt2 = substr($_raw_source, 0, $_loop_seam);
    236. 

  236.           $_snaked_source = trim($_snake_source_pt1 . $_snake_source_pt2);
    237. 

  237.           
    238. 

  238.           # DEBUG
    239. 

  239.           #trigger_notice($_snaked_source);
    240. 

  240.           
    241. 

  241.           # trim by word (see http://www.php.net/manual/en/function.str-word-count.php#59170)
    242. 

  242.           $_BLURB = preg_split("/\s+/", $_snaked_source, ($num_words+1));
    243. 

  243.         unset($_BLURB[(sizeof($_BLURB)-1)]);
    244. 

  244.         $sample =  implode(' ', $_BLURB);
    245. 

  245.           
    246. 

  246.           # finishing touches
    247. 

  247.               
    248. 

  248.         
    249. 

  249.         // RETURN
    250. 

  250.         
    251. 

  251.             return $sample;
    252.

    253.

  391.     function _databug_fetch_lines($source_path)

  392.     {
