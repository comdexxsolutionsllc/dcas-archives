  12.  * Class Node

  13.  * 
    14. 

  14.  * Hierarchical parent/child managing class for database
    15. 

  15.  * 
    16. 

  16.  * Hierarchal data is good for categorizing and structuring data quite similar
    17. 

  17.  * way as xml document is purposed to structure data. This class uses materialized
    18. 

  18.  * path apart from other methods available. Materialized path is quite
    19. 

  19.  * efficient and easy to understand yet it has its limitations because path
    20. 

  20.  * is forced to have varchar, max 255 characters on path field in table.
    21. 

  21.  * However this isn't a big problem until there is a need for very wide hierarchy. If
    22. 

  22.  * we assume 10 000 000 rows in table, there is still room for 36 depth for any
    23. 

  23.  * row.
    24. 

  24.  * 
    25. 

  25.  * To use this class, you need a table with next structure:
    26. 

  26.  * 
    27. 

  27.     CREATE TABLE `treetest` (
    28. 

  28.       `node_id` int(11) NOT NULL auto_increment,

  29.       `node_path` varchar(255) NOT NULL default 'x',
    30. 

  30.       `node_depth` int(11) NOT NULL default '1',
    31. 

  31.       `node_order` int(11) NOT NULL default '0',
    32. 

  32.       `node_title` varchar(255) NOT NULL default '',
    33. 

  33.       `node_has_childs` tinyint(4) NOT NULL default '0',
    34. 

  34.       PRIMARY KEY  (`node_id`),
    35. 

  35.       KEY `node_id` (`node_id`),
    36. 

  36.       KEY `node_path` (`node_path`),
    37. 

  37.       KEY `node_depth` (`node_depth`),
    38. 

  38.       KEY `node_order` (`node_order`),

  39.       KEY `node_title` (`node_title`),
    40. 

  40.       KEY `node_has_childs` (`node_has_childs`)
    41. 

  41.     ) TYPE=MyISAM;
    42. 

  42.  * 
    43. 

  43.  * example rows:
    44. 

  44.  * 
    45. 

  45.     INSERT INTO `treetest` VALUES
    46. 

  46.      ('1','x','1','0','1','1'),

  47.      ('2','x','1','1','2','0'),

  198.     function add_node( $iTo = 0, $sTitle = null, $iOrder = null ) {
    199. 

  199.         if( $iTo > 0 ) {
    200. 

  200.             $oPdo = new PhpDtObject( $this->sTableName, $this->sDsn );
    201. 

  201.             $oPdo->set_mask( array( $this->sNodePath, $this->sNodeDepth, $this->sNodeHasChilds ) );
    202. 

  202.             if( !$oPdo->init( $iTo ) ) {
    203. 

  203.                 //trigger_error( "Could not init node id: $iTo for add_node method." );
    204. 

  204.                 $oPdo->unlink(); unset($oPdo);
    205. 

  205.                 return false;
    206. 

  206.             } else {
    207. 

  207.                 $iToNodeId             = $iTo;
    208. 

  208.                 $iToNodePath         = $oPdo->gv( $this->sNodePath ).$this->sNodePathSeparator.$iTo;
    209. 

  209.                 $iToNodeDepth         = $oPdo->gv( $this->sNodeDepth )+1;
    210. 

  210.                 $iToNodeHasChilds     = $oPdo->gv( $this->sNodeHasChilds );
    211. 

  211.             }
    212. 

  212.         } else {
    213. 

  213.             $iToNodeId         = 0;
    214. 

  214.             $iToNodePath     = $this->sNodePathRoot;
    215. 

  215.             $iToNodeDepth     = 1;
    216. 

  216.         }
    217. 

  217.         // get next order
    218. 

  218.         $iToNodeOrder = isset( $iOrder ) ? $iOrder : $this->_get_next_order($iToNodePath);
    219. 

  219.         $iToNodeTitle = isset( $sTitle ) ? $sTitle : 'Node '.$iToNodeOrder;
    220. 

  220.         // create new node
    221. 

  221.         $oPdo2 = new PhpDtObject( $this->sTableName, $this->sDsn );
    222. 

  222.         $oPdo2->sv( $this->sNodeId, 0 );
    223. 

  223.         $oPdo2->sv( $this->sNodePath, $iToNodePath );
    224. 

  224.         $oPdo2->sv( $this->sNodeDepth, $iToNodeDepth );
    225. 

  225.         $oPdo2->sv( $this->sNodeOrder, $iToNodeOrder );
    226. 

  226.         $oPdo2->sv( $this->sNodeTitle, $iToNodeTitle );
    227. 

  227.         $oPdo2->sv( $this->sNodeHasChilds, 0 );
    228. 

  228.         $iId = $oPdo2->save();
    229. 

  229.         if( $iId ) {
    230. 

  230.             // if node != root and node has no childs, update has_childs field
    231. 

  231.             if( $iToNodeId > 0 && $iToNodeHasChilds == 0 ) {
    232. 

  232.                 $oPdo->sv( $this->sNodeHasChilds, 1 );
    233. 

  233.                 $oPdo->save();
    234. 

  234.             }
    235. 

  235.             // return new node id
    236. 

  236.             $oPdo2->unlink();
    237. 

  237.             if(isset($oPdo)) unset($oPdo);
    238. 

  238.             unset($oPdo2);
    239. 

  239.             return $iId;
    240. 

  240.         } else {
    241. 

  241.             // if insert fails
    242. 

  242.             $oPdo2->unlink(); unset($oPdo2);
    243. 

  243.             if(isset($oPdo)) unset($oPdo);
    244. 

  244.             return false;
    245. 

  245.         }
    246. 

  246.     }

  288.     function delete_node( $iNode, $bOnlySubnodes = false ) {
    289. 

  289.         $oPdo = new PhpDtObject( $this->sTableName, $this->sDsn );
    290. 

  290.         $oPdo->set_mask( array( $this->sNodeId, $this->sNodePath ) );
    291. 

  291.         if( !$oPdo->init($iNode) ) {
    292. 

  292.             if( $iNode > 0 ) {
    293. 

  293.                 //trigger_error( "Could not init node id: $iNode for delete_node method" );
    294. 

  294.                 $oPdo->unlink(); unset($oPdo);
    295. 

  295.                 return false;
    296. 

  296.             } else {
    297. 

  297.                 // if node id is zero (0) then delete all, including autoincrease data!
    298. 

  298.                 $bSuccess = $oPdo->query('TRUNCATE '.$this->sTableName);
    299. 

  299.                 $oPdo->unlink(); unset($oPdo);
    300. 

  300.                 return $bSuccess;
    301. 

  301.             }
    302. 

  302.         }
    303. 

  303.         $aPath         = explode($this->sNodePathSeparator, $oPdo->gv( $this->sNodePath ));
    304. 

  304.         $iParent     = $bOnlySubnodes ? $iNode : $aPath[count($aPath)-1];
    305. 

  305.         $sNodePath     = $oPdo->gv($this->sNodePath).$this->sNodePathSeparator.$iNode;
    306. 

  306.         $sSubnodes     = $bOnlySubnodes ? '' : $iNode > 0 ? " OR ".$this->sNodeId." = '$iNode'" : '';
    307. 

  307.         $sWhere     = $this->sNodePath." = '$sNodePath' OR ".$this->sNodePath." LIKE '$sNodePath.%'$sSubnodes";
    308. 

  308.         $bSuccess     = $oPdo->delete( array( 'where'=>$sWhere ) );
    309. 

  309.         // check parent childs
    310. 

  310.         $this->_update_has_childs($iParent);
    311. 

  311.         $oPdo->unlink(); unset($oPdo);
    312. 

  312.         return $bSuccess;
    313. 

  313.     }

  387.     function copy_node( $iFrom = 0, $iTo = 0, $iDepth = 0, $bOnlySubnodes = false ) {
    388. 

  388.         
    389. 

  389.         $sNodeId         = $this->sNodeId;
    390. 

  390.         $sNodePath         = $this->sNodePath;
    391. 

  391.         $sNodeDepth     = $this->sNodeDepth;

  392.         $sNodeOrder     = $this->sNodeOrder;
    393. 

  393.         $sNodeTitle     = $this->sNodeTitle;

    395. 

  395.         // source node info
    396. 

  396.         $oPdoFrom = new PhpDtObject( $this->sTableName, $this->sDsn );
    397. 

  397.         $oPdoFrom->set_mask( array( $this->sNodeId, $this->sNodePath, $this->sNodeDepth ) );
    398. 

  398.         if( $iFrom > 0 ) {
    399. 

  399.             if( !$oPdoFrom->init($iFrom) ) {
    400. 

  400.                 //trigger_error( "Could not init to node id: $iFrom for copy_node method" );
    401. 

  401.                 return false;
    402. 

  402.             } else {
    403. 

  403.                 $sFromNodePath     = $oPdoFrom->gv($sNodePath).$this->sNodePathSeparator.$iFrom;
    404. 

  404.                 $iFromNodeDepth = $oPdoFrom->gv($sNodeDepth);
    405. 

  405.             }
    406. 

  406.         } else {
    407. 

  407.             $sFromNodePath         = $this->sNodePathRoot;
    408. 

  408.             $iFromNodeDepth     = 0;
    409. 

  409.         }
    410. 

  410.         // target node info
    411. 

  411.         $oPdoTo = new PhpDtObject( $this->sTableName, $this->sDsn );
    412. 

  412.         $oPdoTo->set_mask( array( $this->sNodeId, $this->sNodePath, $this->sNodeDepth, $this->sNodeHasChilds ) );
    413. 

  413.         if( !$oPdoTo->init($iTo) ) {
    414. 

  414.             if( $iTo > 0 ) {
    415. 

  415.                 //trigger_error( "Could not init to node id: $iTo for move_node method" );
    416. 

  416.                 return false;
    417. 

  417.             } else {
    418. 

  418.                 $iToNodePath = $this->sNodePathRoot;
    419. 

  419.             }
    420. 

  420.         } else {
    421. 

  421.             $iToNodePath = $oPdoTo->gv( $sNodePath ).$this->sNodePathSeparator.$iTo;
    422. 

  422.             // update has childs...
    423. 

  423.             if( $iTo > 0 && $oPdoTo->gv( $this->sNodeHasChilds ) == 0 ) {
    424. 

  424.                 $oPdoTo->sv( $this->sNodeHasChilds, 1 );
    425. 

  425.                 $oPdoTo->save();
    426. 

  426.             }
    427. 

  427.         }
    428. 

  428.         // save new nodes
    429. 

  429.         $oPdo1                 = new PhpDtObject( $this->sTableName, $this->sDsn );
    430. 

  430.         $oPdo2                 = new PhpDtObject( $this->sTableName, $this->sDsn );
    431. 

  431.         $aNodes             = array();
    432. 

  432.         // build where clause
    433. 

  433.         $sDepth             = $iDepth > 0 ? ' AND '.$sNodeDepth.' < '.($iFromNodeDepth+$iDepth+1) : '';
    434. 

  434.         $sSubnodes             = $bOnlySubnodes ? '' : $iFrom > 0 ? " OR {$this->sNodeId} = $iFrom" : '';
    435. 

  435.         $sWhere             = "$sNodePath = '$sFromNodePath' OR $sNodePath LIKE '$sFromNodePath.%'$sSubnodes";
    436. 

  436.         $aParam['where']     = '( '.$sWhere.' )'.$sDepth;
    437. 

  437.         $oPdo1->get_many( $aParam );
    438. 

  438.         while( $oPdo1->iterate() ) {
    439. 

  439.             $oPdo2->assign_array( $oPdo1->get_index() );
    440. 

  440.             $oPdo2->sv($sNodeId, 0);
    441. 

  441.             $iId             = $oPdo2->save();
    442. 

  442.             $aReplace[]     = $this->sNodePathSeparator.$iId.$this->sNodePathSeparator;
    443. 

  443.             $aFind[]         = $this->sNodePathSeparator.$oPdo1->gv( $sNodeId ).$this->sNodePathSeparator;
    444. 

  444.             $aNodes[$iId]     = $oPdo1->get_index();
    445. 

  445.         }
    446. 

  446.         // update new nodes with proper path and depth values
    447. 

  447.         if( !empty($aNodes) ) {
    448. 

  448.             // collect update query
    449. 

  449.             $sQuery         = '';
    450. 

  450.             // for path fix...
    451. 

  451.             $sRootFind         = $oPdoFrom->gv( $sNodePath ) != '' ? $bOnlySubnodes ? $oPdoFrom->gv( $sNodePath ).$this->sNodePathSeparator.$iFrom : $oPdoFrom->gv( $sNodePath ) : $this->sNodePathRoot;
    452. 

  452.             $aRootReplace     = $iToNodePath;
    453. 

  453.             foreach( $aNodes as $iId => $aNode ) {
    454. 

  454.                 // path fix
    455. 

  455.                 $sPath         = str_replace( $aFind, $aReplace, $aNode[$sNodePath].$this->sNodePathSeparator );
    456. 

  456.                 $sPath         = str_replace( $sRootFind, $aRootReplace, $sPath );
    457. 

  457.                 $sPath         = rtrim( $sPath, $this->sNodePathSeparator );
    458. 

  458.                 $iDepth     = count( explode( $this->sNodePathSeparator, $sPath ) );
    459. 

  459.                 $iOrder     = $aNode[$sNodeOrder];
    460. 

  460.                 $sTitle     = $aNode[$sNodeTitle];
    461. 

  461.                 $iHasChilds = $aNode[$sNodeHasChilds];
    462. 

  462.                 $sQuery     = "UPDATE `{$this->sTableName}` SET `$sNodePath` = '$sPath', `$sNodeDepth` = $iDepth, `$sNodeOrder` = $iOrder, `$sNodeTitle` = '$sTitle', `$sNodeHasChilds` = $iHasChilds WHERE `$sNodeId` = $iId";
    463. 

  463.                 $oPdo2->query($sQuery);
    464. 

  464.             }
    465. 

  465.         }
    466. 

  466.         return true;
    467. 

  467.     }

  469.     /**
    470. 

  470.      * Move node
    471. 

  471.      * 
    472. 

  472.      * Copies node from source id ($iFrom) to target id ($iTo). 
    473. 

  473.      * If target node is zero (0), then method moves nodes to root.
    474. 

  474.      * Only subnodes argument is used for moving or not moving parent
    475. 

  475.      * node ($iFrom).
    476. 

  476.      * This method checks possible collisions when trying to move node
    477. 

  477.      * into itself.
    478. 

  478.      * 
    479. 

  479.      * @access     public
    480. 

  480.      * @param     integer $iFrom
    481. 

  481.      * @param     integer $iTo default 0
    482. 

  482.      * @param     boolean $bOnlySubnodes default false
    483. 

  483.      * @return     mixed number / boolean
    484. 

  484.      */
    485. 

  485.     function move_node( $iFrom, $iTo = 0, $bOnlySubnodes = false  ) {
    486. 

  486.         // init from & to nodes
    487. 

  487.         $oPdo = new PhpDtObject( $this->sTableName, $this->sDsn );
    488. 

  488.         $oPdo->set_mask( array( $this->sNodeId, $this->sNodePath, $this->sNodeDepth, $this->sNodeHasChilds ) );
    489. 

  489.         if( !$oPdo->init( $iFrom ) ) {
    490. 

  490.             //trigger_error( "Could not init from node id: $iFrom for move_node method" );
    491. 

  491.             return false;
    492. 

  492.         } else {
    493. 

  493.             // parent variable for _update_has_childs
    494. 

  494.             if($bOnlySubnodes){
    495. 

  495.                 $iFromNodeParent = $iFrom;
    496. 

  496.             } else {
    497. 

  497.                 $aPath = explode($this->sNodePathSeparator, $oPdo->gv( $this->sNodePath ));
    498. 

  498.                 $iFromNodeParent = $aPath[count($aPath)-1];
    499. 

  499.             }
    500. 

  500.             // move to itself or if there is no childs available, then
    501. 

  501.             // there is no need for further actions
    502. 

  502.             if( ( $bOnlySubnodes && $oPdo->gv( $this->sNodeHasChilds ) == 0 ) || $iTo == $iFrom ) return true;
    503. 

  503.             // if source node is / is not on root
    504. 

  504.             if( $oPdo->gv( $this->sNodePath ) != $this->sNodePathRoot ) {
    505. 

  505.                 $sFromNodePathWhere = $oPdo->gv( $this->sNodePath ).$this->sNodePathSeparator.$iFrom;
    506. 

  506.                 $sFromNodePathReplace = $oPdo->gv( $this->sNodePath );
    507. 

  507.             } else {
    508. 

  508.                 $sFromNodePathWhere = $this->sNodePathRoot.$this->sNodePathSeparator.$iFrom;
    509. 

  509.                 $sFromNodePathReplace = $this->sNodePathRoot;
    510. 

  510.             }
    511. 

  511.             // variable for depth correction in update clause
    512. 

  512.             if($bOnlySubnodes) {
    513. 

  513.                 $iFromNodeDepth = $oPdo->gv( $this->sNodeDepth );
    514. 

  514.             } else {
    515. 

  515.                 $iFromNodeDepth = $oPdo->gv( $this->sNodeDepth )-1;
    516. 

  516.             }
    517. 

  517.         }
    518. 

  518.         $sFromNodePathWhere1 = $bOnlySubnodes ? $sFromNodePathWhere : $sFromNodePathReplace;
    519. 

  519.         if( !$oPdo->init( $iTo ) ) {
    520. 

  520.             if( $iTo > 0 ) {
    521. 

  521.                 //trigger_error( "Could not init to node id: $iTo for move_node method" );
    522. 

  522.                 return false;
    523. 

  523.             } else {
    524. 

  524.                 $sToNodePathReplace = $this->sNodePathRoot;
    525. 

  525.                 $iToNodeDepth         = 0;
    526. 

  526.             }
    527. 

  527.         } else {
    528. 

  528.             $sToNodePathReplace = $oPdo->gv( $this->sNodePath  ).$this->sNodePathSeparator.$iTo;
    529. 

  529.             $iToNodeDepth         = $oPdo->gv( $this->sNodeDepth );
    530. 

  530.         }
    531. 

  531.         // moving child to first parent wihtout only subnodes needs no further actions
    532. 

  532.         if( $iTo == $iFromNodeParent && !$bOnlySubnodes ) return true;
    533. 

  533.         // moving parent to its child is not possible
    534. 

  534.         if( strpos( $sToNodePathReplace.$this->sNodePathSeparator, $this->sNodePathSeparator.$iFrom.$this->sNodePathSeparator ) ) return false;
    535. 

  535.         // update target node_has_childs...
    536. 

  536.         if( $iTo > 0 && $oPdo->gv( $this->sNodeHasChilds ) == 0 ) {
    537. 

  537.             $oPdo->sv( $this->sNodeHasChilds, 1 );
    538. 

  538.             $oPdo->save();
    539. 

  539.         }
    540. 

  540.         // update paths
    541. 

  541.         $sSubnodes = $bOnlySubnodes ? '' : $iFrom > 0 ? " OR {$this->sNodeId} = $iFrom" : '';
    542. 

  542.         $iDepthCorrection = $iToNodeDepth - $iFromNodeDepth;
    543. 

  543.         $oPdo->query( "UPDATE {$this->sTableName} SET {$this->sNodePath} = REPLACE( {$this->sNodePath}, '$sFromNodePathWhere1', '$sToNodePathReplace' ), {$this->sNodeDepth} = {$this->sNodeDepth} + $iDepthCorrection WHERE {$this->sNodePath} = '$sFromNodePathWhere' OR {$this->sNodePath} LIKE '$sFromNodePathWhere.%'$sSubnodes;" );
    544. 

  544.         // update parent node_has_childs after update paths
    545. 

  545.         $this->_update_has_childs($iFromNodeParent);
    546. 

  546.         return true;
    547. 

  547.     }

  635.     function _fix_orders( $sNodePath ) {

    650.

    651.

  653.             $oPdo->get_many($aParam);
    654. 

  654.             $r=0;
    655.

    657.

    658. 

  658.                 $r++;
    659. 

  659.             }
    660. 

  660.         }

  663.     /**
    664. 

  664.      * Order node up and down helper / factorizing method
    665. 

  665.      * 
    666. 

  666.      * @access     private
    667. 

  667.      * @param     integer $iNode
    668. 

  668.      * @param     string $sMode
    669. 

  669.      * @return  boolean
    670. 

  670.      */
    671. 

  671.     function _order($iNode,$sMode) {
    672. 

  672.         if( $sMode == 'up' ) {
    673. 

  673.             $sAritm1 = '<';
    674. 

  674.             $sAritm2 = '+';
    675. 

  675.             $sOrder = 'DESC';
    676. 

  676.         } else if( $sMode == 'down' ) {
    677. 

  677.             $sAritm1 = '>';
    678. 

  678.             $sAritm2 = '-';
    679. 

  679.             $sOrder = 'ASC';
    680. 

  680.         }
    681. 

  681.         $oPdo = new PhpDtObject( $this->sTableName, $this->sDsn );
    682. 

  682.         $oPdo->set_mask( array( $this->sNodeId, $this->sNodePath, $this->sNodeOrder ) );
    683. 

  683.         if( $oPdo->init($iNode) ) {
    684. 

  684.             // check and fix if there is same order numbers...
    685. 

  685.             $this->_fix_orders( $oPdo->gv( $this->sNodePath ) );
    686. 

  686.             // get node info
    687. 

  687.             $oPdo->init($iNode);
    688. 

  688.             $iOrder1 = $oPdo->gv( $this->sNodeOrder );
    689. 

  689.             $aParam['where'] = "{$this->sNodePath} = '".$oPdo->gv( $this->sNodePath );
    690. 

  690.             $aParam['where'] .= "' AND {$this->sNodeOrder} $sAritm1 ".$oPdo->gv($this->sNodeOrder);
    691. 

  691.             $aParam['order by'] = "{$this->sNodeOrder} $sOrder";
    692. 

  692.             $aParam['limit'] = 1;
    693. 

  693.             // if node is first or last no further actions is needed
    694. 

  694.             $oPdo2 = new PhpDtObject( $this->sTableName, $this->sDsn );
    695. 

  695.             $oPdo2->set_mask( array( $this->sNodeId, $this->sNodePath, $this->sNodeOrder ) );
    696. 

  696.             if( !$oPdo2->get_many($aParam) ) return true;
    697. 

  697.             $oPdo2->iterate();
    698. 

  698.             $iOrder2 = $oPdo2->gv( $this->sNodeOrder );
    699. 

  699.             // calculate cap
    700. 

  700.             if( $sMode == 'up' ) { $iCap = ($iOrder1-1)-$iOrder2; }
    701. 

  701.             else if( $sMode == 'down' ) { $iCap = $iOrder2-$iOrder1-1; }
    702. 

  702.             if( $iCap == 0 ) {
    703. 

  703.                 // next node is order + 1
    704. 

  704.                 $oPdo->sv( $this->sNodeOrder, $iOrder2 );
    705. 

  705.                 $oPdo2->sv( $this->sNodeOrder, $iOrder1 );
    706. 

  706.                 $oPdo2->save();
    707. 

  707.                 $oPdo->save();
    708. 

  708.             } else {
    709. 

  709.                 // next node is order + 1 + n
    710. 

  710.                 $sQuery = "UPDATE {$this->sTableName} SET {$this->sNodeOrder} = {$this->sNodeOrder} $sAritm2 $iCap WHERE {$aParam['where']}";
    711. 

  711.                 $oPdo3 = new PhpDtObject( $this->sTableName, $this->sDsn );
    712. 

  712.                 $oPdo3->query($sQuery);
    713. 

  713.                 $oPdo3->get_many($aParam);
    714. 

  714.                 $oPdo3->iterate();
    715. 

  715.                 $iOrder2 = $oPdo3->gv( $this->sNodeOrder );
    716. 

  716.                 $oPdo->sv( $this->sNodeOrder, $iOrder2 );
    717. 

  717.                 $oPdo3->sv( $this->sNodeOrder, $iOrder1 );
    718. 

  718.                 $oPdo3->save();
    719. 

  719.                 $oPdo->save();
    720. 

  720.                 unset($oPdo3);
    721. 

  721.             }
    722. 

  722.             unset($oPdo2);
    723. 

  723.             $oPdo->unlink(); unset($oPdo);
    724. 

  724.             return true;
    725. 

  725.         }
    726. 

  726.         return false;
    727. 

  727.     }
