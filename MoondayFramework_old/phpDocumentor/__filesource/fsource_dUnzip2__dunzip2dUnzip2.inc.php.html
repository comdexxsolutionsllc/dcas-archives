  291.     Function _loadFileListByEOF(&$fh, $stopOnFile=false){
    292. 

  292.         // Check if there's a valid Central Dir signature.
    293. 

  293.         // Let's consider a file comment smaller than 1024 characters...
    294. 

  294.         // Actually, it length can be 65536.. But we're not going to support it.
    295. 

  295.         
    296. 

  296.         for($x = 0; $x < 1024; $x++){
    297. 

  297.             fseek($fh, -22-$x, SEEK_END);
    298. 

  298.             
    299. 

  299.             $signature = fread($fh, 4);
    300. 

  300.             if($signature == $this->dirSignatureE){
    301. 

  301.                 // If found EOF Central Dir
    302. 

  302.                 $eodir['disk_number_this']   = unpack("v", fread($fh, 2)); // number of this disk
    303. 

  303.                 $eodir['disk_number']        = unpack("v", fread($fh, 2)); // number of the disk with the start of the central directory
    304. 

  304.                 $eodir['total_entries_this'] = unpack("v", fread($fh, 2)); // total number of entries in the central dir on this disk
    305. 

  305.                 $eodir['total_entries']      = unpack("v", fread($fh, 2)); // total number of entries in
    306. 

  306.                 $eodir['size_of_cd']         = unpack("V", fread($fh, 4)); // size of the central directory
    307. 

  307.                 $eodir['offset_start_cd']    = unpack("V", fread($fh, 4)); // offset of start of central directory with respect to the starting disk number
    308. 

  308.                 $zipFileCommentLenght        = unpack("v", fread($fh, 2)); // zipfile comment length
    309. 

  309.                 $eodir['zipfile_comment']    = $zipFileCommentLenght[1]?fread($fh, $zipFileCommentLenght[1]):''; // zipfile comment
    310. 

  310.                 $this->endOfCentral = Array(
    311. 

  311.                     'disk_number_this'=>$eodir['disk_number_this'][1],
    312. 

  312.                     'disk_number'=>$eodir['disk_number'][1],
    313. 

  313.                     'total_entries_this'=>$eodir['total_entries_this'][1],
    314. 

  314.                     'total_entries'=>$eodir['total_entries'][1],
    315. 

  315.                     'size_of_cd'=>$eodir['size_of_cd'][1],
    316. 

  316.                     'offset_start_cd'=>$eodir['offset_start_cd'][1],
    317. 

  317.                     'zipfile_comment'=>$eodir['zipfile_comment'],
    318. 

  318.                 );

  319.                 
    320. 

  320.                 // Then, load file list

  321.                 fseek($fh, $this->endOfCentral['offset_start_cd']);

  322.                 $signature = fread($fh, 4);
    323. 

  323.                 
    324. 

  324.                 while($signature == $this->dirSignature){
    325. 

  325.                     $dir['version_madeby']      = unpack("v", fread($fh, 2)); // version made by
    326. 

  326.                     $dir['version_needed']      = unpack("v", fread($fh, 2)); // version needed to extract
    327. 

  327.                     $dir['general_bit_flag']    = unpack("v", fread($fh, 2)); // general purpose bit flag
    328. 

  328.                     $dir['compression_method']  = unpack("v", fread($fh, 2)); // compression method
    329. 

  329.                     $dir['lastmod_time']        = unpack("v", fread($fh, 2)); // last mod file time
    330. 

  330.                     $dir['lastmod_date']        = unpack("v", fread($fh, 2)); // last mod file date
    331. 

  331.                     $dir['crc-32']              = fread($fh, 4);              // crc-32
    332. 

  332.                     $dir['compressed_size']     = unpack("V", fread($fh, 4)); // compressed size
    333. 

  333.                     $dir['uncompressed_size']   = unpack("V", fread($fh, 4)); // uncompressed size
    334. 

  334.                     $fileNameLength             = unpack("v", fread($fh, 2)); // filename length
    335. 

  335.                     $extraFieldLength           = unpack("v", fread($fh, 2)); // extra field length
    336. 

  336.                     $fileCommentLength          = unpack("v", fread($fh, 2)); // file comment length
    337. 

  337.                     $dir['disk_number_start']   = unpack("v", fread($fh, 2)); // disk number start
    338. 

  338.                     $dir['internal_attributes'] = unpack("v", fread($fh, 2)); // internal file attributes-byte1
    339. 

  339.                     $dir['external_attributes1']= unpack("v", fread($fh, 2)); // external file attributes-byte2
    340. 

  340.                     $dir['external_attributes2']= unpack("v", fread($fh, 2)); // external file attributes
    341.

    342.

    343.

    344.

    345. 

  345.                     
    346. 

  346.                     // Convert the date and time, from MS-DOS format to UNIX Timestamp
    347.

    348. 

  348.                     $BINlastmod_time = str_pad(decbin($dir['lastmod_time'][1]), 16, '0', STR_PAD_LEFT);
    349. 

  349.                     $lastmod_dateY = bindec(substr($BINlastmod_date,  0, 7))+1980;
    350. 

  350.                     $lastmod_dateM = bindec(substr($BINlastmod_date,  7, 4));
    351. 

  351.                     $lastmod_dateD = bindec(substr($BINlastmod_date, 11, 5));
    352. 

  352.                     $lastmod_timeH = bindec(substr($BINlastmod_time,   0, 5));
    353. 

  353.                     $lastmod_timeM = bindec(substr($BINlastmod_time,   5, 6));
    354. 

  354.                     $lastmod_timeS = bindec(substr($BINlastmod_time,  11, 5));    
    355. 

  355.                     
    356. 

  356.                     $this->centralDirList[$dir['file_name']] = Array(
    357. 

  357.                         'version_madeby'=>$dir['version_madeby'][1],
    358. 

  358.                         'version_needed'=>$dir['version_needed'][1],

    360. 

  360.                         'compression_method'=>$dir['compression_method'][1],
    361. 

  361.                         'lastmod_datetime'  =>mktime($lastmod_timeH, $lastmod_timeM, $lastmod_timeS, $lastmod_dateM, $lastmod_dateD, $lastmod_dateY),
    362. 

  362.                         'crc-32'            =>str_pad(dechex(ord($dir['crc-32'][3])), 2, '0', STR_PAD_LEFT).
    363. 

  363.                                               str_pad(dechex(ord($dir['crc-32'][2])), 2, '0', STR_PAD_LEFT).
    364. 

  364.                                               str_pad(dechex(ord($dir['crc-32'][1])), 2, '0', STR_PAD_LEFT).
    365. 

  365.                                               str_pad(dechex(ord($dir['crc-32'][0])), 2, '0', STR_PAD_LEFT),
    366. 

  366.                         'compressed_size'=>$dir['compressed_size'][1],
    367. 

  367.                         'uncompressed_size'=>$dir['uncompressed_size'][1],
    368. 

  368.                         'disk_number_start'=>$dir['disk_number_start'][1],
    369. 

  369.                         'internal_attributes'=>$dir['internal_attributes'][1],
    370. 

  370.                         'external_attributes1'=>$dir['external_attributes1'][1],
    371. 

  371.                         'external_attributes2'=>$dir['external_attributes2'][1],
    372. 

  372.                         'relative_offset'=>$dir['relative_offset'][1],
    373. 

  373.                         'file_name'=>$dir['file_name'],
    374.

  425.     Function _getFileHeaderInformation(&$fh, $startOffset=false){
    426. 

  426.         if($startOffset !== false)
    427. 

  427.             fseek($fh, $startOffset);
    428. 

  428.         
    429. 

  429.         $signature = fread($fh, 4);
    430. 

  430.         if($signature == $this->zipSignature){
    431. 

  431.             # $this->debugMsg(1, "Zip Signature!");
    432. 

  432.             
    433. 

  433.             // Get information about the zipped file
    434. 

  434.             $file['version_needed']     = unpack("v", fread($fh, 2)); // version needed to extract
    435. 

  435.             $file['general_bit_flag']   = unpack("v", fread($fh, 2)); // general purpose bit flag
    436. 

  436.             $file['compression_method'] = unpack("v", fread($fh, 2)); // compression method
    437. 

  437.             $file['lastmod_time']       = unpack("v", fread($fh, 2)); // last mod file time
    438. 

  438.             $file['lastmod_date']       = unpack("v", fread($fh, 2)); // last mod file date
    439. 

  439.             $file['crc-32']             = fread($fh, 4);              // crc-32
    440. 

  440.             $file['compressed_size']    = unpack("V", fread($fh, 4)); // compressed size
    441. 

  441.             $file['uncompressed_size']  = unpack("V", fread($fh, 4)); // uncompressed size
    442. 

  442.             $fileNameLength             = unpack("v", fread($fh, 2)); // filename length
    443. 

  443.             $extraFieldLength           = unpack("v", fread($fh, 2)); // extra field length
    444. 

  444.             $file['file_name']          = fread($fh, $fileNameLength[1]); // filename
    445. 

  445.             $file['extra_field']        = $extraFieldLength[1]?fread($fh, $extraFieldLength[1]):''; // extra field
    446. 

  446.             $file['contents-startOffset']= ftell($fh);
    447. 

  447.             
    448. 

  448.             // Bypass the whole compressed contents, and look for the next file
    449. 

  449.             fseek($fh, $file['compressed_size'][1], SEEK_CUR);
    450. 

  450.             
    451. 

  451.             // Convert the date and time, from MS-DOS format to UNIX Timestamp

  452.             $BINlastmod_date = str_pad(decbin($file['lastmod_date'][1]), 16, '0', STR_PAD_LEFT);

    455. 

  455.             $lastmod_dateM = bindec(substr($BINlastmod_date,  7, 4));
    456. 

  456.             $lastmod_dateD = bindec(substr($BINlastmod_date, 11, 5));
    457. 

  457.             $lastmod_timeH = bindec(substr($BINlastmod_time,   0, 5));
    458. 

  458.             $lastmod_timeM = bindec(substr($BINlastmod_time,   5, 6));
    459. 

  459.             $lastmod_timeS = bindec(substr($BINlastmod_time,  11, 5));
    460. 

  460.             
    461. 

  461.             // Mount file table
    462. 

  462.             $i = Array(
    463. 

  463.                 'file_name'         =>$file['file_name'],
    464. 

  464.                 'compression_method'=>$file['compression_method'][1],

    466.

    468. 

  468.                                       str_pad(dechex(ord($file['crc-32'][2])), 2, '0', STR_PAD_LEFT).
    469.
