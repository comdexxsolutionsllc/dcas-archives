  96.     function http_post ($url, $ipn_notification)
    97. 

  97.     {
    98. 

  98.         // Notification if transfered into a urlencoded string
    99. 

  99.         $post_string = $ipn_notification->to_post_string ()."cmd=_notify-validate";
    100. 

  100.  
    101. 

  101.         if ($connection = curl_init())
    102. 

  102.         {
    103. 

  103.             curl_setopt($connection, CURLOPT_URL, PAYPAL_SECURED_URL);
    104. 

  104.             curl_setopt($connection, CURLOPT_RETURNTRANSFER,1);
    105. 

  105.  
    106. 

  106.             curl_setopt($connection, CURLOPT_POST, 1);
    107. 

  107.             curl_setopt($connection, CURLOPT_POSTFIELDS, $post_string);
    108. 

  108.             
    109. 

  109.             // If you need to go through a proxy server, see set_proxy_options
    110. 

  110.             if(!is_null($this->proxy_url) and !is_null($this->proxy_user_and_pwd))
    111. 

  111.             {
    112. 

  112.                 if(preg_match("/[^:]+:[0-9]+/", $this->proxy_url)
    113. 

  113.                         and preg_match("/([^:]+):.*/", $this->proxy_user_and_pwd, $matches))
    114. 

  114.                 {
    115. 

  115.                     $ipn_notification->add_log ("Proxy settings: user $matches[1] on ".$this->proxy_url);
    116. 

  116.                     curl_setopt($connection, CURLOPT_PROXY, $this->proxy_url);
    117. 

  117.                     curl_setopt($connection, CURLOPT_PROXYUSERPWD, $this->proxy_user_and_pwd);
    118. 

  118.                 }
    119. 

  119.                 else
    120. 

  120.                 {
    121. 

  121.                     $ipn_notification->add_log ("Can't set proxy settings");
    122. 

  122.                 }
    123. 

  123.             }
    124. 

  124.  
    125. 

  125.             $ipn_notification->add_log ("Connection to $url");
    126. 

  126.             $ipn_notification->add_log ("Post string = $post_string");
    127. 

  127.             
    128. 

  128.             $paypal_server_response = curl_exec($connection);
    129. 

  129.             
    130. 

  130.             curl_close($connection);
    131. 

  131.         } 
    132. 

  132.         else
    133. 

  133.         {
    134. 

  134.             $ipn_notification->add_log ("Failed to connect to $url");
    135. 

  135.             return false;
    136. 

  136.         }
    137. 

  137.  
    138. 

  138.         $response = false;
    139. 

  139.         if(preg_match("/(VERIFIED|INVALID)/", $paypal_server_response, $matches))
    140. 

  140.         {
    141. 

  141.             $response = $matches[1];
    142. 

  142.         }
    143. 

  143.  
    144. 

  144.         return $response;
    145. 

  145.     }
